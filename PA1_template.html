Are there differences in activity patterns between weekdays and weekends?

Use wday() and bind_cols() to create a new column weekdays

library(lubridate)
x <- sub('2', 'weekday',
        sub('3', 'weekday',
            sub('4', 'weekday',
                sub('5', 'weekday',
                    sub('6', 'weekday',
                        sub('7', 'weekend', 
                            sub('1', 'weekend',
                                wday(new$date))))))))
weekdays <- bind_cols(new, data.table(weekdays = x))
